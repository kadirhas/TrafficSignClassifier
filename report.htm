    def normalize_grayscale(image_data):
        """
        Normalize the image data with Min-Max scaling to a range of [-1, 1]
        :param image_data: The image data to be normalized
        :return: Normalized image data
        """
        a = -1
        b = 1
        grayscale_min = 0
        grayscale_max = 255
        return a + ( ( (image_data - grayscale_min)*(b - a) )/( grayscale_max - grayscale_min ) )

    def LeNet(x):    
        # Arguments used for tf.truncated_normal, randomly defines variables for the weights and biases for each layer
        mu = 0
        sigma = 0.1
        
        
        # TODO: Layer 1: Convolutional. Input = 32x32x1. Output = 28x28x6.
        w1 = tf.Variable(tf.truncated_normal(shape = [5, 5, 1, 6],mean = mu, stddev = sigma))
        b1 = tf.Variable(tf.zeros(6))
        c1 = tf.nn.conv2d(x, w1, strides=[1, 1, 1, 1], padding='VALID')
        c1 = tf.nn.bias_add(c1, b1)
        # TODO: Activation.
        c1 = tf.nn.relu(c1)
        # TODO: Pooling. Input = 28x28x6. Output = 14x14x6.
        c1 = tf.nn.max_pool(c1, ksize=[1, 2, 2, 1], strides=[1, 2, 2, 1], padding='VALID')
        # TODO: Layer 2: Convolutional. Output = 10x10x16.
        w2 = tf.Variable(tf.truncated_normal(shape = [5, 5, 6, 16], mean = mu, stddev = sigma))
        b2 = tf.Variable(tf.zeros(16))
        c2 = tf.nn.conv2d(c1, w2, strides=[1, 1, 1, 1], padding='VALID')
        c2 = tf.nn.bias_add(c2, b2)
        # TODO: Activation.
        c2 = tf.nn.relu(c2)
        # TODO: Pooling. Input = 10x10x16. Output = 5x5x16.
        c2 = tf.nn.max_pool(c2, ksize=[1, 2, 2, 1], strides=[1, 2, 2, 1], padding='VALID')
    ​
        # 1x1 convolution layer
        cw1 = tf.Variable(tf.truncated_normal(shape = [5, 5, 16, 400], mean = mu, stddev = sigma))
        cb1 = tf.Variable(tf.zeros(400))
        c2 = tf.nn.conv2d(c2, cw1, strides=[1, 1, 1, 1], padding='VALID')
        c2 = tf.nn.bias_add(c2, cb1)
        
        # TODO: Flatten. Input = 5x5x16. Output = 400.
        c2 = flatten(c2)
        c2 = tf.nn.dropout(c2, keep_prob)
        # TODO: Layer 3: Fully Connected. Input = 400. Output = 120.
        fw1 = tf.Variable(tf.truncated_normal(shape = [400, 120], mean = mu, stddev = sigma))
        fb1 = tf.Variable(tf.zeros(120))
        fc1 = tf.add(tf.matmul(c2, fw1), fb1)
        # TODO: Activation.
        fc1 = tf.nn.relu(fc1)
        fc2 = tf.nn.dropout(fc1, keep_prob)
        # TODO: Layer 4: Fully Connected. Input = 120. Output = 100.
        fw2 = tf.Variable(tf.truncated_normal(shape = [120, 100], mean = mu, stddev = sigma))
        fb2 = tf.Variable(tf.zeros(100))
        fc2 = tf.add(tf.matmul(fc2, fw2), fb2)
        # TODO: Activation.
        fc3 = tf.nn.relu(fc2)
        fc3 = tf.nn.dropout(fc3, keep_prob)
        # TODO: Layer 5: Fully Connected. Input = 84. Output = 10.
        fw3 = tf.Variable(tf.truncated_normal(shape = [100, 43], mean = mu, stddev = sigma))
        fb3 = tf.Variable(tf.zeros(43))
        fc3 = tf.add(tf.matmul(fc3, fw3), fb3)
        logits = fc3
        
        return logits

    def evaluate(X_data, y_data):
        num_examples = len(X_data)
        total_accuracy = 0
        sess = tf.get_default_session()
        for offset in range(0, num_examples, BATCH_SIZE):
            batch_x, batch_y = X_data[offset:offset+BATCH_SIZE], y_data[offset:offset+BATCH_SIZE]
            accuracy = sess.run(accuracy_operation, feed_dict={x: batch_x, y: batch_y, keep_prob: 1.0})
            total_accuracy += (accuracy * len(batch_x))
        return total_accuracy / num_examples
